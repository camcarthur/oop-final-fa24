228

257                        )

259                        self._db_session.add_all(
260                            [from_transaction, to_transaction])
261
262                        logger.debug(f"After Transfer - From Account Balance: {from_account.balance}, To Account Balance: {to_account.balance}")
263
264                        self._db_session.commit()

266
267                    elif transfer_type == "external":
268                        to_account_id_str = request.form.get('toExternalAccount', '0')
269                        to_account_id = int(to_account_id_str)
270                        to_account = self._db_session.query(Account).filter_by(
271                            account_id=to_account_id).first()
272
273                        if not to_account:
274                            logger.error("Invalid external target account.")
275                            return make_response("Invalid external target account.", 400)
276
277                        # Process external
278                        from_account.balance -= amount
279                        to_account.balance += amount
280
281                        # Create transaction records
282                        from_transaction = Transaction(
283                            account_id=from_account_id,
284                            transaction_type=TransactionType.withdrawal,
285                            amount=amount,
286                            status=TransactionStatus.completed
287                        )
288                        to_transaction = Transaction(
289                            account_id=to_account_id,
290                            transaction_type=TransactionType.deposit,
291                            amount=amount,
292                            status=TransactionStatus.completed
293                        )
294
295                        self._db_session.add_all(
296                            [from_transaction, to_transaction])
297
298                        # Commit transaction
299                        self._db_session.commit()
300                        logger.info(f"External Transfer: ${amount} from {from_account.account_id} to {to_account.account_id}")
301
302                    return redirect(url_for('dashboard'))
303
304                except Exception as e:
305                    self._db_session.rollback()
306                    logger.exception("An error occurred during the transfer.")
307                    return make_response(f"An error occurred during the transfer: {str(e)}", 500)
308
309            # Fetch users
310            user_accounts = self._db_session.query(Account).filter_by(
311                user_id=user.user_id).all()
312            return render_template('transfer.html', accounts=user_accounts)

416def login_required(f: Callable[..., Any]) -> Callable[..., Any]:
417    """
418    Decorator to check if the user is logged in
419    """
420    @wraps(f)
421    def decorated_function(*args: Any, **kwargs: Any) -> Any:
422        if 'user_id' not in session:
423            return redirect(url_for('login'))
424        return f(*args, **kwargs)
425    return decorated_function

428def admin_required(f: Callable[..., Any]) -> Callable[..., Any]:
429    """
430    Decorator to check if the user is an admin before accessing certain routes.
431    """
432    @wraps(f)
433    def decorated_function(*args: Any, **kwargs: Any) -> Any:
434        user_id = session.get('user_id')
435        user: Optional[User] = bank_app.db_session.query(User).filter_by(
436            user_id=user_id).first()
437        if not user or user.role != Role.admin:
438            return redirect(url_for('login'))
439        return f(*args, **kwargs)
440    return decorated_function
